import random import math import asyncio from js import document, console, setTimeout # Game configuration INITIAL_POINTS = 100 SECTIONS = 12 WIN_PROBABILITY = 0.1 # Only 10% chance to win # Multipliers - more negative options to favor the house MULTIPLIERS = [ -5, -4, -3, -3, -2, -2, -1, -1, -1, 2, 3, 5 ] class SpinGame: def __init__(self): self.points = INITIAL_POINTS self.is_spinning = False self.history = [] self.setup_wheel() def setup_wheel(self): wheel = document.getElementById("wheel") wheel.innerHTML = "" # Create wheel sections for i in range(SECTIONS): section = document.createElement("div") section.className = "wheel-section" # Calculate angle and color angle = 360 / SECTIONS * i hue = angle % 360 bg_color = f"hsl({hue}, 70%, 50%)" text_color = "white" if hue < 60 or hue > 240 else "black" # Set styles and content section.style.transform = f"rotate({angle}deg)" section.style.backgroundColor = bg_color section.style.color = text_color section.textContent = f"{MULTIPLIERS[i]}x" # Alternate text position for readability if i % 2 == 0: section.style.justifyContent = "flex-start" section.style.paddingLeft = "20px" section.style.paddingBottom = "20px" else: section.style.justifyContent = "flex-end" section.style.paddingRight = "20px" section.style.paddingTop = "20px" wheel.appendChild(section) async def spin(self): if self.is_spinning or self.points <= 0: return self.is_spinning = True spin_btn = document.getElementById("spinBtn") spin_btn.disabled = True result_div = document.getElementById("result") result_div.textContent = "" result_div.className = "result" # Determine the outcome (90% chance of losing) will_win = random.random() < WIN_PROBABILITY if will_win: # Find a positive multiplier index possible_indices = [i for i, m in enumerate(MULTIPLIERS) if m > 0] winning_index = random.choice(possible_indices) multiplier = MULTIPLIERS[winning_index] else: # Find a negative multiplier index possible_indices = [i for i, m in enumerate(MULTIPLIERS) if m < 0] losing_index = random.choice(possible_indices) multiplier = MULTIPLIERS[losing_index] # Calculate the final rotation (5 full rotations + offset to selected section) section_angle = 360 / SECTIONS target_angle = 5 * 360 + (360 - (winning_index if will_win else losing_index) * section_angle - section_angle / 2) # Apply the spin animation wheel = document.getElementById("wheel") wheel.style.transform = f"rotate({target_angle}deg)" # Wait for the spin to complete await asyncio.sleep(4) # Calculate new points new_points = self.points + multiplier * 10 # Using 10 as the base stake if new_points < 0: new_points = 0 # Update points and display result self.points = new_points document.getElementById("points").textContent = str(self.points) # Show result with animation result_text = f"{'+' if multiplier > 0 else ''}{multiplier}x! New points: {self.points}" result_div.textContent = result_text if multiplier > 0: result_div.classList.add("win") else: result_div.classList.add("lose") # Add to history self.add_to_history(multiplier) # Reset for next spin self.is_spinning = False spin_btn.disabled = False # Disable button if no points left if self.points <= 0: spin_btn.disabled = True result_div.textContent = "Game Over! Refresh to play again." def add_to_history(self, multiplier): history_div = document.getElementById("history") item = document.createElement("div") item.className = f"history-item {'win-history' if multiplier > 0 else 'lose-history'}" item.textContent = f"{'Win' if multiplier > 0 else 'Lose'}: {multiplier}x (Points: {self.points})" # Insert at the top if history_div.children.length > 1: history_div.insertBefore(item, history_div.children[1]) else: history_div.appendChild(item) # Initialize the game game = SpinGame() # Set up the spin button def spin_wheel(event): asyncio.ensure_future(game.spin()) spin_btn = document.getElementById("spinBtn") spin_btn.addEventListener("click", spin_wheel)
